CREATE TABLE korisnik (
    Id BIGINT(255) NOT NULL AUTO_INCREMENT,
    Korisnik_id BIGINT(255) NOT NULL,
    Korisnik VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Grad VARCHAR(255) CHARACTER SET cp1250 COLLATE cp1250_croatian_ci NOT NULL,
    Mobitel VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Sifra VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Date TIMESTAMP(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    ImeSlika VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Slika VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    PRIMARY KEY (Id),
    INDEX Korisnik_id (Korisnik_id),
    INDEX Korisnik (Korisnik),
    INDEX Grad (Grad),
    INDEX Mobitel (Mobitel),
    INDEX Sifra (Sifra),
    INDEX ImeSlika (ImeSlika),
    INDEX Slika (Slika)
);

CREATE TABLE messages (
    id  INT(11) AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci,
    Profilna VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    message TEXT CHARACTER SET latin2 COLLATE latin2_croatian_ci,
    timestamp TIMESTAMP DEFAULT current_timestamp()
);
CREATE TABLE vjesti (
    Izvodac VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Zanr VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Datum DATE NOT NULL,
    ClubLok VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Opis VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Lokacija VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Cijena INT(255) NOT NULL
);
CREATE TABLE objave(
    id INT(255) AUTO_INCREMENT PRIMARY KEY,
    Korisnik VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Profilna VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Slika VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Naslov VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_croatian_ci NOT NULL,
    Komentar VARCHAR(255) CHARACTER SET latin2 COLLATE latin2_croatian_ci NOT NULL,
    Datum TIMESTAMP DEFAULT current_timestamp() ON UPDATE CURRENT_TIMESTAMP()
);
